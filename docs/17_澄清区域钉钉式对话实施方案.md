## 四、新增工具函数（clarifyUtils.js）

新建 `js/clarifyUtils.js`，提供全局函数：

```javascript
(function (global) {
  'use strict';
  function formatClarifyTime(iso) {
    if (!iso) return '';
    var d = new Date(iso);
    var now = new Date();
    var today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    var msgDate = new Date(d.getFullYear(), d.getMonth(), d.getDate());
    var h = ('0' + d.getHours()).slice(-2);
    var m = ('0' + d.getMinutes()).slice(-2);
    var time = h + ':' + m;
    if (msgDate.getTime() === today.getTime()) return time;
    if (msgDate.getTime() === today.getTime() - 86400000) return '昨天 ' + time;
    return (d.getMonth() + 1) + '月' + d.getDate() + '日 ' + time;
  }
  function roleAvatar(role) {
    return { operator: '运', supplier: '供', verifier: '核' }[role] || '?';
  }
  global.formatClarifyTime = formatClarifyTime;
  global.roleAvatar = roleAvatar;
})(typeof window !== 'undefined' ? window : global);
```

各页面需在 `<script>` 前引入：

```html
<script src="../js/clarifyUtils.js"></script>
```

## 五、页面内新增 JS 逻辑

在各 `task_detail_*.html` 的 `setup()` 内新增/扩展：

### 5.1 当前用户角色

```javascript
const currentClarifyRole = 'operator'; // 运营端固定；供应商端为 'supplier'；认证端为 'verifier'
```

### 5.2 气泡左右判断

```javascript
function isSelf(role) {
  return role === currentClarifyRole;
}
```

### 5.3 时间分组线

```javascript
function shouldShowTimeDivider(idx) {
  var msgs = selectedClarDetail.value && selectedClarDetail.value.messages || [];
  if (idx === 0) return true;
  var prev = msgs[idx - 1];
  var curr = msgs[idx];
  if (!prev || !curr) return false;
  return (new Date(curr.createdAt) - new Date(prev.createdAt)) > 5 * 60 * 1000;
}
```

### 5.4 导出

在 `return` 中增加：

```javascript
formatClarifyTime: typeof formatClarifyTime === 'function' ? formatClarifyTime : function(t) { return t || ''; },
roleAvatar: typeof roleAvatar === 'function' ? roleAvatar : function(r) { return '?'; },
isSelf, shouldShowTimeDivider
```
